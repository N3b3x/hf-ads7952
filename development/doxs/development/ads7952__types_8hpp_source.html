177template <typename... Channels>
178constexpr uint16_t ChannelMask(Channels... channels) noexcept {
179 static_assert((std::is_integral_v<Channels> && ...),
180 "ChannelMask arguments must be integers");
181 uint16_t mask = 0;
182 ((mask |= static_cast<uint16_t>(1U << (static_cast<uint8_t>(channels) & 0x0F))), ...);
183 return mask & 0x0FFF;
184}

187constexpr uint16_t ChannelRangeMask(uint8_t first, uint8_t last) noexcept {
188 if (first > 11 || last > 11 || first > last) return 0;
189 // (1 << (last+1)) - (1 << first) generates contiguous set bits
190 return static_cast<uint16_t>(
191 ((1U << (last + 1)) - (1U << first)) & 0x0FFF);
192}
