148 // Alarm group mode for a given global channel number (0-11)
149 inline constexpr uint16_t GroupMode(uint8_t channel) noexcept {
150 uint8_t group = channel / 4;
151 constexpr uint16_t modes[] = {
152 Mode::ALARM_GROUP_0, Mode::ALARM_GROUP_1,
153 Mode::ALARM_GROUP_2, Mode::ALARM_GROUP_3
154 };
155 return (group < 4) ? modes[group] : Mode::ALARM_GROUP_0;
